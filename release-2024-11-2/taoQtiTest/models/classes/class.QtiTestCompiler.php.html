313    /**
314     * Compile a QTI Test and the related QTI Items.
315     *
316     * The compilation process occurs as follows:
317     *
318     * * 1. The resources composing the test are copied into the private compilation directory.
319     * * 2. The test definition is packed (test and items put together in a single definition).
320     * * 3. The items composing the test are compiled.
321     * * 4. The rubric blocks are rendered into PHP templates.
322     * * 5. The test definition is compiled into PHP source code for maximum performance.
323     * * 6. The resources composing the test that have to be accessed at delivery time are compied into the public
324     *      compilation directory.
325     * * 7. The Service Call definition enabling TAO to run the compiled test is built.
326     *
327     * @return tao_models_classes_service_ServiceCall A ServiceCall object that represent the way to call the newly
328     *                                                compiled test.
329     * @throws taoQtiTest_models_classes_QtiTestCompilationFailedException If an error occurs during the compilation.
330     */
331    public function compile()
332    {
333
334        $report = new common_report_Report(common_report_Report::TYPE_INFO);
335
336        try {
337            // 0. Initialize compilation (compilation directories, renderers, ...).
338            $this->initCompilation();
339
340            // 1. Copy the resources composing the test into the private complilation directory.
341            $this->copyPrivateResources();
342
343            // 2. Compact the test definition itself.
344            $compiledDoc = $this->compactTest();
345
346            // 3. Compile the items of the test.
347            $itemReport = $this->compileItems($compiledDoc);
348            $report->add($itemReport);
349            if ($itemReport->getType() != common_report_Report::TYPE_SUCCESS) {
350                common_Logger::e($report->getMessage(), $report->getErrors());
351                $msg = 'Failed item compilation.';
352                $code = taoQtiTest_models_classes_QtiTestCompilationFailedException::ITEM_COMPILATION;
353                throw new taoQtiTest_models_classes_QtiTestCompilationFailedException(
354                    $msg,
355                    $this->getResource(),
356                    $code
357                );
358            }
359
360            // 4. Explode the rubric blocks in the test into rubric block refs.
361            $this->explodeRubricBlocks($compiledDoc);
362
363            // 5. Update test definition with additional runtime info.
364            $assessmentTest = $compiledDoc->getDocumentComponent();
365            //$this->updateTestDefinition($assessmentTest);
366
367            // 6. Compile rubricBlocks and serialize on disk.
368            $this->compileRubricBlocks($assessmentTest);
369
370            // 7. Copy the needed files into the public directory.
371            $this->copyPublicResources();
372

810     * Remote resources can be refereced by the following QTI classes/attributes:
811     *

813     * * object:data
814     * * img:src
815     *
816     * @param AssessmentTest $assessmentTest An AssessmentTest object.
817     * @throws taoQtiTest_models_classes_QtiTestCompilationFailedException If a remote resource cannot be retrieved.
818     */
819    protected function copyRemoteResources(RubricBlock $rubricBlock)
820    {
821        $ds = DIRECTORY_SEPARATOR;
822        $tmpDir = tao_helpers_File::createTempDir();
823        $destPath = trim($this->getExtraPath(), $ds) . $ds
824            . taoQtiTest_models_classes_QtiTestService::TEST_REMOTE_FOLDER . $ds;
825
826        // Search for all class-attributes in QTI-XML that might reference a remote file.
827        $search = $rubricBlock->getComponentsByClassName(['a', 'object', 'img']);
828        foreach ($search as $component) {
829            switch ($component->getQtiClassName()) {
830                case 'object':
831                    $url = $component->getData();
832                    break;
833
834                case 'img':
835                    $url = $component->getSrc();
836                    break;
837            }
838
